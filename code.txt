class Customer :public Shop {
private:
    int quant[20];
    string buy[20];
    int i=0,total=0;
public:
    void buy_item();
    void calculate();
};

void Customer::buy_item() {
    cout << "\nWhat do you want to buy?" << endl;
    while (buy[i] != "DONE") {
        cin >> buy[i];
        if (buy[i] == "0")
            break;
        cout << "Enter quantity: ";
        cin >> quant[i];
        i++;
        cout << "\nAnother item: " << endl;
    }
}

void Customer :: calculate() {
    view_list.open("Items.txt", ios::in);
    if (view_list.is_open()) {
        while (!view_list.fail()) {
            view_list >> item_code >> item_name >> brand_name >> price >> quantity;
            for (i = 0; i < 20; i++) {
                if (buy[i] == item_code) {
                    total = total + (price * quant[i]);
                    break;
                }
            }
        }
        view_list.close();
    }
    else {
        cout << "FILE NOT FOUND" << endl;
    }
    cout << "Total=" << total;
}
