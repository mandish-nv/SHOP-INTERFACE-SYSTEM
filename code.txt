void Admin::remove_item()  {
    string option;
    view_list.open("Items.txt", ios::in);
    temp_file.open("temp.txt", ios::out);
    cout << "Enter code number of item to remove: ";
    cin >> option;
    if (view_list.is_open()) {
        while (!view_list.eof()) {
            view_list >> item_code >> item_name >> brand_name >> price >> quantity;
            if (item_code != option) {
                temp_file << " " << item_code << " " << item_name << " " << brand_name << " " << price << " " << quantity;
            }
            else if (item_code == option) {
                continue;
            }
        }
        view_list.close();
        temp_file.close();
        remove("Items.txt");
        rename("temp.txt", "Items.txt");
    }
    else {
        cout << "FILE NOT FOUND" << endl;
    }
}
